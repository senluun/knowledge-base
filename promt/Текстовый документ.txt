    <script>
        // Инициализация темы
        function initTheme() {
            const savedTheme = localStorage.getItem('theme') || 'dark';
            document.documentElement.setAttribute('data-bs-theme', savedTheme);
            updateThemeIcon(savedTheme);
        }
        
        // Инициализация dropdown меню
        function initDropdowns() {
            // Проверяем, что Bootstrap загружен
            if (typeof bootstrap === 'undefined') {
                console.warn('Bootstrap не загружен, повторная попытка через 100ms...');
                setTimeout(initDropdowns, 100);
                return;
            }
            
            const dropdownToggleList = [].slice.call(document.querySelectorAll('[data-bs-toggle="dropdown"]'));
            dropdownToggleList.map(function (dropdownToggleEl) {
                const dropdown = new bootstrap.Dropdown(dropdownToggleEl, {
                    autoClose: false,
                    boundary: 'viewport'
                });
                
                // Предотвращаем закрытие при клике внутри меню
                dropdownToggleEl.addEventListener('click', function(e) {
                    e.stopPropagation();
                });
                
                // Предотвращаем закрытие при клике на dropdown-menu
                const dropdownMenu = dropdownToggleEl.nextElementSibling;
                if (dropdownMenu) {
                    dropdownMenu.addEventListener('click', function(e) {
                        e.stopPropagation();
                    });
                }
                
                return dropdown;
            });
        }
        
        // Обновление иконки темы
        function updateThemeIcon(theme) {
            const themeIcon = document.getElementById('themeIcon');
            if (theme === 'dark') {
                themeIcon.className = 'fas fa-moon';
            } else {
                themeIcon.className = 'fas fa-sun';
            }
        }
        
        // Переключение темы
        function toggleTheme() {
            const currentTheme = document.documentElement.getAttribute('data-bs-theme');
            const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
            
            document.documentElement.setAttribute('data-bs-theme', newTheme);
            localStorage.setItem('theme', newTheme);
            updateThemeIcon(newTheme);
        }
        
        // Функции для toast уведомлений
        function showToast(message, type = 'info') {
            const toast = document.getElementById('toast');
            const toastBody = toast.querySelector('.toast-body');
            const toastHeader = toast.querySelector('.toast-header');
            const icon = toastHeader.querySelector('i');
            
            // Устанавливаем содержимое
            toastBody.textContent = message;
            
            // Устанавливаем иконку и цвет в зависимости от типа
            icon.className = 'fas me-2';
            switch(type) {
                case 'success':
                    icon.classList.add('fa-check-circle', 'text-success');
                    break;
                case 'error':
                case 'danger':
                    icon.classList.add('fa-exclamation-circle', 'text-danger');
                    break;
                case 'warning':
                    icon.classList.add('fa-exclamation-triangle', 'text-warning');
                    break;
                default:
                    icon.classList.add('fa-info-circle', 'text-primary');
            }
            
            // Показываем toast
            const bsToast = new bootstrap.Toast(toast);
            bsToast.show();
        }
        
        // Глобальная функция для показа уведомлений
        window.showNotification = showToast;
        
        // Инициализация при загрузке
        document.addEventListener('DOMContentLoaded', function() {
            initTheme();
            initDropdowns();
            
            // Обработчик кнопки переключения темы
            document.getElementById('themeToggle').addEventListener('click', toggleTheme);
        });
    </script>
